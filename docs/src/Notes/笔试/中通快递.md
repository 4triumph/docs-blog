### 手写 Promise 实现（异步编程）
要求：
**then() 方法：**

接受两个参数：
- 第一个参数是一个函数，用于处理 Promise resolved 时的结果。
- 第二个参数也是一个函数，用于处理 Promise rejected 时的错误信息。
- then() 方法需要支持链式调用。
**catch() 方法：**
用于处理 Promise rejected 的错误信息。它等价于 then(null, onRejected)，只处理 Promise 的失败情况。

示例 1：
```javascript
const promise = new MyPromise((resolve) => {
  setTimeout(() => {
    resolve('success');
  }, 100);
});

promise.then(value => {
  console.log(value);
});
// 输出：success

```
示例 2：
```javascript
const promise = new MyPromise((resolve, reject) => {
  setTimeout(() => {
    reject(new Error('failure'));
  }, 100);
});

promise.catch(reason => {
  console.log(reason.message);
});
// 输出：failure
```

代码
```javascript
class MyPromise {
  constructor(executor) {
    this.state = 'pending'; // 初始状态为 pending
    this.value = undefined; // 成功时的返回值
    this.reason = undefined; // 失败时的原因

    this.onResolvedCallbacks = []; // 存储成功的回调
    this.onRejectedCallbacks = []; // 存储失败的回调

    // resolve 函数
    const resolve = (value) => {
      if (this.state === 'pending') {
        this.state = 'resolved'; // 状态变为 resolved
        this.value = value;
        // 执行所有成功的回调
        this.onResolvedCallbacks.forEach(fn => fn());
      }
    };

    // reject 函数
    const reject = (reason) => {
      if (this.state === 'pending') {
        this.state = 'rejected'; // 状态变为 rejected
        this.reason = reason;
        // 执行所有失败的回调
        this.onRejectedCallbacks.forEach(fn => fn());
      }
    };

    try {
      // 执行传入的 executor 函数
      executor(resolve, reject);
    } catch (err) {
      // 如果出错，直接执行 reject
      reject(err);
    }
  }

  // then 方法
  then(onFulfilled, onRejected) {
    onFulfilled = typeof onFulfilled === 'function' ? onFulfilled : value => value;
    onRejected = typeof onRejected === 'function' ? onRejected : reason => { throw reason };

    // 返回新的 promise 支持链式调用
    return new MyPromise((resolve, reject) => {
      if (this.state === 'resolved') {
        // 异步执行
        setTimeout(() => {
          try {
            const x = onFulfilled(this.value);
            resolve(x);
          } catch (e) {
            reject(e);
          }
        });
      } else if (this.state === 'rejected') {
        // 异步执行
        setTimeout(() => {
          try {
            const x = onRejected(this.reason);
            reject(x);
          } catch (e) {
            reject(e);
          }
        });
      } else if (this.state === 'pending') {
        // 如果还处于 pending 状态，将回调函数保存到数组中
        this.onResolvedCallbacks.push(() => {
          setTimeout(() => {
            try {
              const x = onFulfilled(this.value);
              resolve(x);
            } catch (e) {
              reject(e);
            }
          });
        });

        this.onRejectedCallbacks.push(() => {
          setTimeout(() => {
            try {
              const x = onRejected(this.reason);
              reject(x);
            } catch (e) {
              reject(e);
            }
          });
        });
      }
    });
  }

  // catch 方法
  catch(onRejected) {
    return this.then(null, onRejected);
  }
}
```
